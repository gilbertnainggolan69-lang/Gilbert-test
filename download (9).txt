        const canvas = document.getElementById('bg-particles');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth; canvas.height = window.innerHeight;
        let particlesArray = [];

        class Particle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.size = Math.random() * 1.2 + 0.2;
                this.speedX = (Math.random() - 0.5) * 0.2;
                this.speedY = (Math.random() - 0.5) * 0.2;
                this.color = `rgba(0, 153, 255, ${Math.random() * 0.15})`;
            }
            update() {
                this.x += this.speedX;
                this.y += this.speedY;
                if(this.x < 0 || this.x > canvas.width) this.speedX *= -1;
                if(this.y < 0 || this.y > canvas.height) this.speedY *= -1;
            }
            draw() {
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function init() {
            particlesArray = [];
            for(let i = 0; i < 100; i++) particlesArray.push(new Particle());
        }

        function animate() {
            ctx.clearRect(0,0,canvas.width,canvas.height);
            particlesArray.forEach(p => { p.update(); p.draw(); });
            requestAnimationFrame(animate);
        }
        init(); animate();
        window.onresize = () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; init(); };


       
        async function process() {
            const u = document.getElementById('url').value;
            if(!u) { alert("Masukkan alamat URL terlebih dahulu!"); return; }
            
            const res = document.getElementById('result');
            const out = document.getElementById('out');
            const statusBox = document.getElementById('statusBox');
            const statusText = document.getElementById('statusText');
            const progressBar = document.getElementById('progressBar');
            const loadingIcon = document.getElementById('loadingIcon');

            res.style.display = 'block';
            res.classList.add('opacity-100');
            statusBox.style.display = 'block';
            statusText.textContent = "Memproses permintaan... Menghubungkan ke server...";
            statusText.className = "text-yellow-400 font-medium";
            loadingIcon.classList.remove('hidden');
            progressBar.style.width = "10%";
            out.value = "// SISTEM ANALISIS AKTIF\n// Target: " + u + "\n// By Gilbert Tools\n\n";

            
            const proxies = [
                `https://api.allorigins.win/raw?url=${encodeURIComponent(u)}`,
                `https://corsproxy.io/?${encodeURIComponent(u)}`,
                `https://proxy.cors.sh/${u}`,
                `https://cors-anywhere.herokuapp.com/${u}`,
                `https://thingproxy.freeboard.io/fetch/${u}`,
                `https://www.whateverorigin.org/get?url=${encodeURIComponent(u)}&callback=`,
                `https://crossorigin.me/${u}`,
                `https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(u)}`,
                `https://api.proxiesite.com/proxy?url=${encodeURIComponent(u)}`,
                `https://api.allorigins.win/raw?url=${encodeURIComponent(u)}`
            ];

            let sukses = false;
            let logError = "";

            for (let i = 0; i < proxies.length; i++) {
                try {
                    let persen = 10 + (i * 9);
                    progressBar.style.width = persen + "%";
                    statusText.textContent = `Memindai struktur... [Koneksi ${i+1}/10]`;
                    
                    const response = await fetch(proxies[i], {
                        headers: { 'X-Requested-With': 'XMLHttpRequest' }
                    });
                    
                    if (!response.ok) throw new Error("Koneksi gagal");
                    
                    const data = await response.text();
                    
                    progressBar.style.width = "100%";
                    statusText.textContent = "✅ Berhasil! Data kode telah dimuat sepenuhnya.";
                    statusText.className = "text-green-400 font-medium";
                    out.value = `/* ==============================================
   HASIL ANALISIS KODE SUMBER
   Alamat: ${u}
   Waktu: ${new Date().toLocaleString()}
   Dibuat oleh: GILBERT
============================================== */

${data}`;
                    sukses = true;
                    break;
                } catch (err) {
                    logError += `⚠️ Server ${i+1}: Tidak tersedia\n`;
                    continue;
                }
            }

            if(!sukses){
                progressBar.style.width = "100%";
                progressBar.style.background = "#F53F3F";
                statusText.textContent = "❌ Gagal mengakses! Target memiliki proteksi tinggi atau link salah.";
                statusText.className = "text-red-400 font-medium";
                out.value = `⚠️ PEMBERITAHUAN SISTEM ⚠️

Sistem tidak dapat mengakses alamat tersebut.

Kemungkinan:
1. Website memiliki proteksi keamanan tingkat tinggi.
2. Alamat URL salah atau tidak lengkap.
3. Koneksi tidak stabil.

— Gilbert Professional Tools`;
            }
            loadingIcon.classList.add('hidden');
        }

        
        function copy() {
            const out = document.getElementById('out');
            out.select();
            document.execCommand('copy');
            const btn = document.getElementById('copyBtn');
            btn.innerHTML = "✅ Disalin!";
            btn.classList.add('bg-green-600');
            setTimeout(() => {
                btn.innerHTML = "<i class='fa-solid fa-copy mr-2'></i> Salin Kode";
                btn.classList.remove('bg-green-600');
            }, 2000);
        }

        
        function clearAll() {
            document.getElementById('url').value = "";
            document.getElementById('result').style.display = "none";
            document.getElementById('result').classList.remove('opacity-100');
            document.getElementById('statusBox').style.display = "none";
        }

        
        function saveAsFile() {
            const content = document.getElementById('out').value;
            if(!content) { alert("Data belum tersedia! Silakan analisis dulu."); return; }
            const blob = new Blob([content], {type: 'text/html'});
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `Gilbert_Code_${new Date().getTime()}.html`;
            a.click();
            URL.revokeObjectURL(url);
        }
